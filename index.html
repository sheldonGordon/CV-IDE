public class Experience{
  private String entreprise;
  private String ville;
  private String poste;
  private LocalDate dateDebut;
  private LocalDate dateFin;
  private String description;

  public Experience(String entreprise, String ville, String poste, LocalDate dateDebut,
                    LocalDate dateFin, String description){
    this.setEntreprise(entreprise);
    this.setVille(ville);
    this.setPoste(poste);
    this.setDateDebut(dateDebut);
    this.setDateFin(dateFin);
    this.setDescription(description);
  }

  public List< Experience > getListExperience(){
    List< Experience > resultList = new ArrayList< Experience >(3);

    resultList.add(new Experience("Ministère des Armées",
                                  "Paris",
                                  "Développeur JAVA/JEE",
                                  LocalDate.of(2011, 10, 1),
                                  LocalDate.of(2015, 8, 1),
                                  "Développement d'application web en respectant le cahier des charges, " +
                                  "avec l'ensemble des nouvelles technologies (EJB3, JPA, ...)."));

    return resultList;
  }

  public String getEntreprise(){
    return this.entreprise;
  }

  public void setEntreprise(String entreprise){
    this.entreprise = entreprise;
  }

  public String getVille(){
    return this.ville;
  }

  public void setVille(String ville){
    this.ville = ville;
  }

  public String getPoste(){
    return this.poste;
  }

  public void setPoste(String poste){
    this.poste = poste;
  }

  public LocalDate getDateDebut(){
    return this.dateDebut;
  }

  public void setDateDebut(LocalDate dateDebut){
    this.dateDebut = dateDebut;
  }

  public LocalDate getDateFin(){
    return this.dateFin;
  }

  public void setDateFin(LocalDate dateFin){
    this.dateFin = dateFin;
  }
}
